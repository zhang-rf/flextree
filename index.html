FlextreeJS is a lightweight open source jQuery plugin that provides flexible tree-like layout such as treeview, it's template-based and it's fully customizable.
Get Started Fork me

Getting Started

  1. Setup a container
    <div id="demo"></div>
  2. Include jQuery
  3. Download FlextreeJS and include flextree.js
    <script src="flextree.js"></script>
  4. Define a template
    You can use placeholders in template, they will be automatically replaced during generation
    var template = '<a href="javascript:;">{myAwesomeText}</a>';
  5. Create a data source
    You can get data source by ajax, here just simulate a data source
    var dataSource = [{id: 'A1', parent_id: null, myAwesomeText: 'I am A1', myAwesomeData: 'AwesomeA1'},
    {id: 'A2', parent_id: 'A1', myAwesomeText: 'I am A2', myAwesomeData: 'AwesomeA2'},
    {id: 'B1', parent_id: null, myAwesomeText: 'I am B1', myAwesomeData: 'AwesomeB1'},
    {id: 'C1', parent_id: null, myAwesomeText: 'I am C1', myAwesomeData: 'AwesomeC1'}];
  6. Finally, create your tree
    Non-collapsible tree
    $('#demo').tree(dataSource, template);
    Collapsible tree with default collapse event: onClick
    $('#demo').tree(dataSource, template, {collapsible: true});
    Custom nodes during binding
    $('#demo').tree(dataSource, template,
    function (data, i) {
        // This will be called each time a node generated
        if (i == 0)
            $(this).append(' I feel good');

        if (data.id == 'C1')
            $(this).children().append(' I also feel good');

        $(this).data('awesome', data.myAwesomeData).on('click', function () {
            alert($(this).children().text() + ': ' + $(this).data('awesome'));
        });
    });
  7. Complete code
    <div id="demo"></div>

<script>
    $(function () {
        var template = '<a href="javascript:;">{myAwesomeText}</a>';
        var dataSource = [{id: 'A1', parent_id: null, myAwesomeText: 'I am A1', myAwesomeData: 'AwesomeA1'},
            {id: 'A2', parent_id: 'A1', myAwesomeText: 'I am A2', myAwesomeData: 'AwesomeA2'},
            {id: 'B1', parent_id: null, myAwesomeText: 'I am B1', myAwesomeData: 'AwesomeB1'},
            {id: 'C1', parent_id: null, myAwesomeText: 'I am C1', myAwesomeData: 'AwesomeC1'}];

        $('#demo').tree(dataSource, template,
            function (data, i) {
                // This will be called each time a node generated
                if (i == 0)
                    $(this).append(' I feel good');

                if (data.id == 'C1')
                    $(this).children().append(' I also feel good');

                $(this).data('awesome', data.myAwesomeData).on('click', function () {
                    alert($(this).children().text() + ': ' + $(this).data('awesome'));
                });
        });
    });
</script>
